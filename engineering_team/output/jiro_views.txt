### 2. Stimulus Controllers

**TicketController**
```javascript
// app/javascript/controllers/ticket_controller.js
import { Controller } from "stimulus";

export default class extends Controller {
  static targets = ["status", "priority", "assignee"];

  connect() {
    // initialize filters here if needed
  }

  changeStatus(event) {
    this.performAction(event, 'transition');
  }

  changeAssignee(event) {
    this.performAction(event, 'assign');
  }

  performAction(event, action) {
    event.preventDefault();
    const url = event.target.dataset.url;
    const value = event.target.value;

    fetch(url, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'X-CSRF-Token': document.querySelector('meta[name="csrf-token"]').getAttribute('content')
      },
      body: JSON.stringify({ action, value })
    }).then(response => {
      if (response.ok) {
        // handle success, e.g., display a modal or update the UI
      } else {
        // handle error
      }
    });
  }
}
```

### 4. Turbo Frames & Streams

**Ticket Details with Comments**
```slim
-# app/views/tickets/show.html.slim
.container
  h1 = @ticket.title
  p = @ticket.description

  .ticket-details
    h3 Details
    p Status: 
      = render 'components/status_badge', status: @ticket.status
    p Priority: 
      = @ticket.priority.titleize
    p Assignee: 
      = @ticket.assigned_to&.name || 'Unassigned'

    = render 'components/assignment_dropdown', ticket: @ticket

  hr

  h3 Comments
  = turbo_frame_tag dom_id(@ticket, :comments) do
    = render @ticket.comments

  = render 'comments/form', ticket: @ticket
```
